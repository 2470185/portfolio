const processPrompt = async (prompt) => {
  const result = await gemini
    .generateContent({
      contents: [{
        role: 'user',
        parts: [{ text: prompt }]
      }]
    });
  // stream into WebContainer
  await mountToContainer(
    result.generatedFiles
  );

    files: result.generatedFiles
  });
